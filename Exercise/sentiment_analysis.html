데이터 전처리#

  • nsmc 데이터 중 아래와 같이 document 컬럼 데이터가 없는 row가 존재합니다.

  • 비어있는 데이터 처리는 [1] 평균 값 혹은 정책에 따라 적절한 값으로 대체하거나 [2] 삭제 하는 방법이 있습니다.

  • 본 예제에서는 편의를 위해 [2] 삭제 (dropna) 방식으로 처리할 예정입니다.

  • ‘Train, Test Data 생성’ 과정을 통해 생성된 데이터를 로딩하는 과정입니다

def preprocessing(df):
  # 빈 값 제거
  df = df.dropna(subset=['document'])
  # 중복 제거
  df = df.drop_duplicates(subset=['document'])
  return df

def load_prep_dataset(path):
  df = load_raw_dataset(path)  
  df_pref = preprocessing(df)
  print(f"total dataset count: {len(df)}")
  print(f"after preprocessing, dataset count: {len(df_pref)}")
  return df_pref

from transformers import AdamW

optimizer = AdamW(model.parameters(), lr=5e-5)

from transformers import get_scheduler

num_epochs = 3
num_training_steps = num_epochs * len(tr_dataloader)
lr_scheduler = get_scheduler(
    "linear",
    optimizer=optimizer,
    num_warmup_steps=0,
    num_training_steps=num_training_steps
)

model.to(args.device)

from tqdm.auto import tqdm

progress_bar = tqdm(range(num_training_steps))

model.train()
for epoch in range(num_epochs):
    for batch in tr_dataloader:
        # print(batch)
        batch = {k: v.to(args.device) for k, v in batch.items()}
        outputs = model(**batch)
        loss = outputs.loss
        loss.backward()

        optimizer.step()
        lr_scheduler.step()
        optimizer.zero_grad()
        progress_bar.update(1)
